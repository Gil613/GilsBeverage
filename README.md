# GilsBeverage

~~~sql
CREATE TABLE GIL_PRODUCT(
    IDX NUMBER NOT NULL PRIMARY KEY,
    FILENAME VARCHAR2(50),
    PNAME NVARCHAR2(50),
    PCONTENT NVARCHAR2(100),
    PTYPE NVARCHAR2(10),
    CAPACITY VARCHAR2(10),
    COST NUMBER
);
    CREATE SEQUENCE PRODUCT_IDX_SEQUENCE 
    START WITH 1
    INCREMENT BY 1;


CREATE TABLE GIL_USER(
    ID NVARCHAR2(20) NOT NULL PRIMARY KEY,
    PWD NVARCHAR2(30) NOT NULL,
    P_SHA256 NVARCHAR2(100) NOT NULL,
    P_BCRYPT NVARCHAR2(100) NOT NULL,
    NAME NVARCHAR2(10) NOT NULL,
    PHONE VARCHAR2(30),
    EMAIL NVARCHAR2(50) NOT NULL,
    REGDATE DATE DEFAULT SYSDATE
);

CREATE TABLE GIL_BOARD(
    IDX NUMBER NOT NULL PRIMARY KEY,
    ID NVARCHAR2(20) NOT NULL,
    FILENAME NVARCHAR2(100),
    REGDATE DATE DEFAULT SYSDATE,
    TITLE NVARCHAR2(30),
    CONTENT NVARCHAR2(1000),
    FLAG NVARCHAR2(10) NOT NULL,
    CONSTRAINT FK_USERID FOREIGN KEY(ID) REFERENCES GIL_USER(ID)
               ON DELETE CASCADE
);
    CREATE SEQUENCE BOARD_IDX_SEQUENCE 
    START WITH 1
    INCREMENT BY 1;
    
    Drop Table Gil_Product;
    Drop Table Gil_User;
    Drop Table Gil_Board;
    
    DROP SEQUENCE BOARD_IDX_SEQUENCE;
    DROP SEQUENCE PRODUCT_IDX_SEQUENCE;
    
    CREATE TABLE GIL_SCRAP(
    IDX NUMBER NOT NULL PRIMARY KEY,
    ID NVARCHAR2(20) NOT NULL,
    PNAME NVARCHAR2(50),
    REGDATE DATE DEFAULT SYSDATE,
    CONSTRAINT FK_SCRAPID FOREIGN KEY(ID) REFERENCES GIL_USER(ID)
               ON DELETE CASCADE
);
~~~
